311  // Displays a desired text string on the display.

313  // @param text String to write on the display.
314  // @param position Position to start writing the characters
315  void DisplayText(const char * text,
316  CursorPosition_t position = kDefaultCursorPosition) override
317  {
318  constexpr uint8_t kMaxDisplayWidth = 20;
319  const uint8_t kStartOffset = position.position;
320  uint8_t termination_index = uint8_t(strlen(text));
321  // calculate the termination_index to stop writing to the display
322  // if the string length of text exceeds the display's width
323  if (termination_index > kMaxDisplayWidth)
324  {
325  termination_index = kMaxDisplayWidth;
326  }
327  if (termination_index > (kMaxDisplayWidth - kStartOffset))
328  {
329  termination_index = uint8_t(kMaxDisplayWidth - kStartOffset);
330  }
331  // set cursor start position for writing
332  SetCursorPosition(position);
333  for (uint8_t i = 0; i < termination_index; i++)
334  {
335  WriteData(text[i]);
336  }
337  }
