149  uint32_t CalculatePll(PllInput input_frequency, uint16_t desired_speed_in_mhz)
150  {
151  SJ2_ASSERT_FATAL(desired_speed_in_mhz < 384 && desired_speed_in_mhz > 12,
152  "Frequency must be lower than 384 MHz"
153  "and greater than or equal to 12 MHz");
154  bool calculating = true;
155  uint32_t multiplier_value;
156  if ((desired_speed_in_mhz % static_cast<uint16_t>(input_frequency)) >= 1)
157  {
158  multiplier_value = static_cast<uint32_t>(
159  (desired_speed_in_mhz / static_cast<uint16_t>(input_frequency)) + 1);
160  }
161  else
162  {
163  multiplier_value = static_cast<uint32_t>(
164  desired_speed_in_mhz / static_cast<uint16_t>(input_frequency));
165  }
166  uint16_t divider_value = 1;
167  while (calculating)
168  {
169  uint16_t current_controlled_oscillator_frequency;
170  current_controlled_oscillator_frequency = static_cast<uint16_t>(
171  (static_cast<uint16_t>(input_frequency) * multiplier_value * 2) *
172  divider_value);
173  if (current_controlled_oscillator_frequency >= 156)
174  {
175  calculating = false;
176  }
177  else
178  {
179  divider_value = static_cast<uint16_t>(divider_value * 2);
180  SJ2_ASSERT_FATAL(divider_value < 8,
181  "PLL divider value went out of bounds");
182  }
183  }
184 
185  return multiplier_value;
186  }

188  uint32_t SetMainPll(PllInput input_frequency, uint16_t desired_speed_in_mhz)
189  {
190  uint16_t divider_value = 1;
191  uint64_t timeout_time = Milliseconds() + kDefaultTimeout;
192  uint64_t current_time = Milliseconds();
193  uint32_t multiplier_value =
194  CalculatePll(input_frequency, desired_speed_in_mhz);
195  uint32_t actual_speed =
196  static_cast<uint32_t>(input_frequency) * multiplier_value;
197  // TO DO: use registers to retreive values
198  SelectOscillatorSource(OscillatorSource::kIrc);
199  SelectMainClockSource(MainClockSource::kBaseClock);
200  SelectUsbClockSource(UsbSource::kBaseClock);
201  SelectSpifiClockSource(SpifiSource::kBaseClock);
202  // must subtract 1 from multiplier value as specified in datasheet
203  LPC_SC->PLL0CFG =
204  (LPC_SC->PLL0CFG & ~kClearPllMultiplier) | (multiplier_value - 1);
205  LPC_SC->PLL0CFG =
206  (LPC_SC->PLL0CFG & ~kClearPllDivider) | (divider_value << 5);
207  LPC_SC->PLL0CON |= kEnablePll;
208  // nessecary feed sequence to ensure the changes are intentional
209  LPC_SC->PLL0FEED = 0xAA;
210  LPC_SC->PLL0FEED = 0x55;
211  while (!(LPC_SC->PLL0STAT >> kPlock & 1) && (current_time < timeout_time))
212  {
213  current_time = Milliseconds();
214  }
215  if (!(LPC_SC->PLL0STAT >> kPlock & 1) && (current_time >= timeout_time))
216  {
217  SJ2_ASSERT_FATAL(false,
218  "PLL lock could not be established before timeout");
219  }
220 
221  return (actual_speed - desired_speed_in_mhz);
