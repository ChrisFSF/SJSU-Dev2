168  // Powers on the desired peripheral and sets the peripheral clock. Also,
169  // enables the SSP data lines, format, data size, and more. See page 601
170  // of user manual UM10562 LPC408x/407x for more details.
171  bool Initialize() override
172  {
173  bool peripheral_initialized;
174  uint32_t pssp = static_cast<uint32_t>(pssp_);
175 
176  if (pssp_ > Peripheral::kSsp1)
177  {
178  peripheral_initialized = 0;
179  }
180  else
181  {
182  // if frame mode is set for Microwire, force data_size_ = 8 bits
183  DataSize check_data_size =
184  (frame_format_ == kMicro) ? kEight : data_size_;
185  // Power up peripheral
186  sysclock_register->PCONP &= ~(1 << pconp_bits_);
187  sysclock_register->PCONP |= (1 << pconp_bits_);
188  ssp_registers[pssp]->CPSR &= ~(0xFF);
189  ssp_registers[pssp]->CPSR |= (clock_prescaler_);
190 
191  // Enable SSP pins
192  mosi_.SetPinFunction(kEnableSspPins);
193  miso_.SetPinFunction(kEnableSspPins);
194  sck_.SetPinFunction(kEnableSspPins);
195 
196  // Clear and Set Control Register values
197  ssp_registers[pssp]->CR0 &=
198  ~((0xF << kDataBit) | (0x3 << kFrameBit) | (0x3 << kPolarityBit) |
199  (0xFF << kDividerBit));
200  ssp_registers[pssp]->CR0 |=
201  (check_data_size << kDataBit) | (frame_format_ << kFrameBit);
202 
203  // set clk polarity = 1, clk phase set to 0
204  ssp_registers[pssp]->CR0 |=
205  (clock_polarity_ << kPolarityBit) | (clock_phase_ << kPhaseBit);
206  ssp_registers[pssp]->CR0 |= (clock_divider_ << kDividerBit);
207  // enable SSP
208  ssp_registers[pssp]->CR1 &= ~(1 << 1);
209  ssp_registers[pssp]->CR1 |= (1 << 1);
210  ssp_registers[pssp]->CR1 &= ~(1 << kMasterModeBit);
211  ssp_registers[pssp]->CR1 |= (master_mode_ << kMasterModeBit);
212  peripheral_initialized = 1;
213  }
214  return peripheral_initialized;
215  }
