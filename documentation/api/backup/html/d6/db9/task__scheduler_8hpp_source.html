59  // not be added and a fatal error will be asserted.
60  //
61  // @note When a task inheriting the TaskInterface is constructed, it will
62  // automatically call this function to add itself to the scheduler.
63  //
64  // @param task Task to add.
65  void AddTask(TaskInterface * task) override
66  {

69  "The scheduler is currently full, the task will not be "

71  for (uint8_t i = 0; i < config::kTaskSchedulerSize; i++)
72  {
73  if (task_list_[i] == nullptr)
74  {
75  task_list_[i] = task;
76  task_count_++;
77  return;
78  }
79  }
