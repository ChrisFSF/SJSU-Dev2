191  int CalculateTemperature(I2c temperature_i2c)
192  {
193  // Enables one burst mode (on-demand) temperature sampling
194  // by setting control register 0xC4 oneburst bit 3 to 1
195  // and stop bit 2 to 0. Stop bit is set to 1 upon successful
196  // measurement.
197  temperature_i2c.Write(kTemperatureAddress, { 0xC4, 0x04 });
198 
199  // Enable the auto increment of the i2c register address pointer
200  // by setting control register 0xC5 to autoinc bit to 1
201  temperature_i2c.Write(kTemperatureAddress, { 0xC5, 0x01 });
202  // MSB, LSB for temperature data
203  uint8_t temperature_ms = 0xC1;
204  uint8_t temperature_ls = 0xC2;
205  int temperature_data = 0;
206  temperature_i2c.WriteThenRead(kTemperatureAddress, &temperature_ms, 1,
207  &temperature_ms, 1);
208  temperature_i2c.Read(kTemperatureAddress, &temperature_ls, 1);
209 
210  temperature_data = ((temperature_ms & 0x7F) << 8) | temperature_ls;
211  printf(" Temperature Data: 0x%02X, 0x%02X, 0x%04X\n\n", temperature_ms,
212  temperature_ls, temperature_data);
213  // Compute the actual temperature in Celsius
214  return (55 + ((temperature_data - 16384) / 160));
215  }
