658  std::pair<iterator, bool> insert(const value_type& key)
659  {
660  std::pair<iterator, bool> result(end(), false);
661 
662  ETL_ASSERT(!full(), ETL_ERROR(unordered_set_full));
663 
664  // Get the hash index.
665  size_t index = get_bucket_index(key);
666 
667  // Get the bucket & bucket iterator.
668  bucket_t* pbucket = pbuckets + index;
669  bucket_t& bucket = *pbucket;
670 
671  // The first one in the bucket?
672  if (bucket.empty())
673  {
674  // Get a new node.
675  node_t& node = *pnodepool->allocate<node_t>();
676  ::new (&node.key) value_type(key);
677  ETL_INCREMENT_DEBUG_COUNT
678 
679  // Just add the pointer to the bucket;
680  bucket.insert_after(bucket.before_begin(), node);
681 
682  result.first = iterator(pbuckets + number_of_buckets, pbucket, pbucket->begin());
683  result.second = true;
684 
685  adjust_first_last_markers(pbucket);
686  }
687  else
688  {
689  // Step though the bucket looking for a place to insert.
690  local_iterator inode_previous = bucket.before_begin();
691  local_iterator inode = bucket.begin();
692 
693  while (inode != bucket.end())
694  {
695  // Do we already have this key?
696  if (inode->key == key)
697  {
698  break;
699  }
700 
701  ++inode_previous;
702  ++inode;
703  }
704 
705  // Not already there?
706  if (inode == bucket.end())
707  {
708  // Get a new node.
709  node_t& node = *pnodepool->allocate<node_t>();
710  ::new (&node.key) value_type(key);
711  ETL_INCREMENT_DEBUG_COUNT
712 
713  // Add the node to the end of the bucket;
714  bucket.insert_after(inode_previous, node);
715  ++inode_previous;
716 
717  result.first = iterator(pbuckets + number_of_buckets, pbucket, inode_previous);
718  result.second = true;
719  }
720  }
