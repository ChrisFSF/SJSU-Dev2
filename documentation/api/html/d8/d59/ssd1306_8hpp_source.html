180  void DrawPixel(int32_t x, int32_t y, Color_t color) final override
181  {
182  // The 3 least significant bits hold the bit position within the byte
183  uint32_t bit_position = y & 0b111;
184  // Each byte makes up a vertical column.
185  // Shifting by 3, which also divides by 8 (the 8-bits of a column), will
186  // be the row that we need to edit.
187  uint32_t row = y >> 3;
188  // Mask to clear the bit
189  uint32_t clear_mask = ~(1 << bit_position);
190  // Mask to set the bit, if color.alpha != 0
191  bool pixel_is_on = (color.alpha != 0);
192  uint32_t set_mask = pixel_is_on << bit_position;
193  // Address of the pixel column to edit
194  uint8_t * pixel_column = &(bitmap_[row][x]);
195  // Read pixel column and update the pixel
196  uint32_t result = (*pixel_column & clear_mask) | set_mask;
197  // Update pixel with the result of this operation
198  *pixel_column = static_cast<uint8_t>(result);
199  }
