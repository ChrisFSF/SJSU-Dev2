182  void SetPixel(size_t x, size_t y, bool pixel_is_on = true)
183  {
184  // The 3 least significant bits hold the bit position within the byte
185  uint32_t bit_position = y & 0b111;
186  // Each byte makes up a vertical column.
187  // Shifting by 3, which also divides by 8 (the 8-bits of a column), will
188  // be the row that we need to edit.
189  uint32_t row = y >> 3;
190  // Mask to clear the bit
191  uint32_t clear_mask = ~(1 << bit_position);
192  // Mask to set the bit, if pixel_is_on = true
193  uint32_t set_mask = pixel_is_on << bit_position;
194  // Address of the pixel column to edit
195  uint8_t * pixel_column = &(bitmap_[row][x]);
196  // Read pixel column and update the pixel
197  uint32_t result = (*pixel_column & clear_mask) | set_mask;
198  // Write back resulting operation
199  *pixel_column = static_cast<uint8_t>(result);
200  }
