209  template <typename TReturn, const size_t NBITS, typename TValue>
210  TReturn fold_bits(TValue value)
211  {
212  ETL_STATIC_ASSERT(integral_limits<TReturn>::bits >= NBITS, "Return type too small to hold result");
213 
214  const TValue mask = etl::power<2, NBITS>::value - 1;
215  const size_t shift = NBITS;
216 
217  // Fold the value down to fit the width.
218  TReturn folded_value = 0;
219 
220  // Keep shifting down and XORing the lower bits.
221  while (value >= etl::max_value_for_nbits<NBITS>::value)
222  {
223  folded_value ^= value & mask;
224  value >>= shift;
225  }
226 
227  // Fold the remaining bits.
228  folded_value ^= value & mask;
229 
230  return folded_value;
231  }
