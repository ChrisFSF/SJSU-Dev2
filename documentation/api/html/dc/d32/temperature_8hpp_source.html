42  {
43  uint8_t most_significant_register;
44  uint8_t least_significant_register;
45  // The register will enable the device to collect data once
46  // and automatically sets the stop bit to 0 (2nd bit).
47  i2c_->Write(kDeviceAddress, { kOneBurstRegister, 0x04 }, 2);
48  // Auto increments I2c register address pointer.
49  i2c_->Write(kDeviceAddress, { kAutonicBitRegister, 0x01 }, 2);
50  i2c_->WriteThenRead(kDeviceAddress, { kMostSignificantRegister },
51  &most_significant_register, 1);
52  i2c_->Read(kDeviceAddress, &least_significant_register, 1);
53 
54  uint16_t temperature_data = 0;
55  // Clears the most significant bit from above and shifts all bit.
56  // The write and read operation sets most significant bit to one,
57  // telling the register that the data has been read.
58  most_significant_register = bit::Clear(most_significant_register, 7);
59  temperature_data =
60  bit::Insert(temperature_data, most_significant_register, 8, 16);
61  temperature_data =
62  bit::Insert(temperature_data, least_significant_register, 0, 8);
63  // The required computation after bit shifting.
64  // Formula can be found below, see page 4:
65  // datasheets/Temperature-Sensor/si7060-datasheets.pdf
66  // Note that: 1 << 14 = 2^14 = 16384
67  return (((temperature_data - (1 << 14)) / 160) + 55);
68  }
