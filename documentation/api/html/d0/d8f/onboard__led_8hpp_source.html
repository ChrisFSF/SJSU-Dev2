85  // This function takes in the 4 least significant bits from value, and sets
86  // the led to be ON or OFF. The least significant bit corresponds to LED0,
87  // next least significant corresponds to LED1, etc. ON =1, OFF =0. The four
88  // most significant bits will be unused.
89  void SetAll(uint8_t value) override
90  {
91  for (uint8_t i = 0; i < 4; i++)
92  {
93  if ((value >> i) & 1)
94  {
95  Set(i, LightState::kOn);
96  }
97  else
98  {
99  Set(i, LightState::kOff);
100  }
101  }
102  }
103  // This function will return the states of LED0-LED3, going from bit 0
104  // represnting LED 0, to bit 3 representing LED3. Only the four least
105  // significant bits will be used. The four most significant bits will be 0s.
106  uint8_t GetStates(void) override
107  {
108  uint32_t led_states = 0x0000;
109  for (uint8_t i = 0; i < 4; i++)
110  {
111  led_states |= led[i].Read() << i;
112  }
113  return static_cast<uint8_t>(led_states);
114  }
