38 [[gnu::always_inline]][[nodiscard]]
39 constexpr T Extract(T target, uint32_t position, uint32_t width = 1)
40 {
41  // Check the types at compile time
42  static_assert(std::numeric_limits<T>::is_integer,
43  "Extract only accepts intergers.");
44  // Need to use an unsigned version of the type T for the mask to make sure
45  // that the shift right doesn't result in a sign extended shift.
46  using UnsignedT = typename std::make_unsigned<T>::type;
47  // At compile time, generate variable containing all 1s with the size of the
48  // target parameter.
49  constexpr UnsignedT kFieldOfOnes = GenerateFieldOfOnes<T>();
50  // At compile time calculate the number of bits in the target parameter.
51  constexpr size_t kTargetWidth = sizeof(T) * 8;
52  // Create mask by shifting the set of 1s down so that the number of 1s from
53  // bit position 0 is equal to the width parameter.
54  UnsignedT mask = kFieldOfOnes >> (kTargetWidth - width);
55  // Shift target down to the right to get the bit position in the 0th bit
56  // location.
57  // Mask the value to clear any bit after the width's amount of bits.
58  return (target >> position) & mask;
59 }

95  constexpr UnsignedT kFieldOfOnes = GenerateFieldOfOnes<T>();

98  // Create mask by shifting the set of 1s down so that the number of 1s from
