173  //
174  // @param operation Operation transfer type.

176  void WriteNibble(WriteOperation operation, uint8_t nibble) override
177  {
178  kControlPins.e.SetHigh();
179  kControlPins.rs.Set(Gpio::State(operation));

312  // Displays a desired text string on the display.
313  //
314  // @param text String to write on the display.
315  // @param position Position to start writing the characters
316  void DisplayText(const char * text,
317  CursorPosition_t position = kDefaultCursorPosition) override
318  {
319  constexpr uint8_t kMaxDisplayWidth = 20;
320  const uint8_t kStartOffset = position.position;
321  uint8_t termination_index = uint8_t(strlen(text));
322  // calculate the termination_index to stop writing to the display
323  // if the string length of text exceeds the display's width
324  if (termination_index > kMaxDisplayWidth)
325  {
326  termination_index = kMaxDisplayWidth;
327  }
328  if (termination_index > (kMaxDisplayWidth - kStartOffset))
329  {
330  termination_index = uint8_t(kMaxDisplayWidth - kStartOffset);
331  }
332  // set cursor start position for writing
333  SetCursorPosition(position);
334  for (uint8_t i = 0; i < termination_index; i++)
335  {
336  WriteData(text[i]);
337  }
338  }
