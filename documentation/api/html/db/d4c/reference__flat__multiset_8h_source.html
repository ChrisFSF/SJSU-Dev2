462  std::pair<iterator, bool> insert(value_type& value)
463  {
464  std::pair<iterator, bool> result(end(), false);
465 
466  ETL_ASSERT(!lookup.full(), ETL_ERROR(flat_multiset_full));
467 
468  iterator i_element = std::lower_bound(begin(), end(), value, compare);
469 
470  if (i_element == end())
471  {
472  // At the end. Doesn't exist.

474  result.first = --end();
475  result.second = true;
476  }
477  else
478  {
479  // Not at the end.
480  lookup.insert(i_element.ilookup, &value);
481  result.first = i_element;
482  result.second = true;
483  }
484 
485  return result;
