360  //*************************************************************************
361  char* allocate_item()
362  {
363  char* p_value = nullptr;
364 
365  // Any free space left?
366  if (items_allocated < MAX_SIZE)
367  {
368  // Initialise another one if necessary.
369  if (items_initialised < MAX_SIZE)
370  {
371  uintptr_t p = reinterpret_cast<uintptr_t>(p_buffer + (items_initialised * ITEM_SIZE));
372  *reinterpret_cast<uintptr_t*>(p) = p + ITEM_SIZE;
373  ++items_initialised;
374  }
375 
376  // Get the address of new allocated item.
377  p_value = p_next;
378 
379  ++items_allocated;
380  if (items_allocated != MAX_SIZE)
381  {
382  // Set up the pointer to the next free item
383  p_next = *reinterpret_cast<char**>(p_next);
384  }
385  else
386  {
387  // No more left!
388  p_next = nullptr;
389  }
390  }
391  else
392  {
393  ETL_ASSERT(false, ETL_ERROR(pool_no_allocation));
394  }
395 
396  return p_value;

402  void release_item(char* p_value)
403  {
404  // Does it belong to us?
405  ETL_ASSERT(is_item_in_pool(p_value), ETL_ERROR(pool_object_not_in_pool));
406 
407  if (p_next != nullptr)
408  {

411  }
412  else
413  {
414  // This is the only free item.

426  bool is_item_in_pool(const char* p) const
427  {
428  // Within the range of the buffer?
429  intptr_t distance = p - p_buffer;
430  bool is_within_range = (distance >= 0) && (distance <= intptr_t((ITEM_SIZE * MAX_SIZE) - ITEM_SIZE));
431 
432  // Modulus and division can be slow on some architectures, so only do this in debug.
433 #if defined(ETL_DEBUG)
434  // Is the address on a valid object boundary?
435  bool is_valid_address = ((distance % ITEM_SIZE) == 0);
436 #else
437  bool is_valid_address = true;
438 #endif
