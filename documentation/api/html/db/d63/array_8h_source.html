438  {
439  iterator p = const_cast<iterator>(position);
440  iterator result(p);
441 
442  size_t source_size = std::distance(first, last);
443  size_t destination_space = std::distance(position, cend());
444 
445  // Do we need to move anything?
446  if (source_size < destination_space)
447  {
448  size_t length = SIZE - (std::distance(begin(), p) + source_size);
449  std::copy_backward(p, p + length, end());
