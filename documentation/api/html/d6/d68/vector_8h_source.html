636  void insert(iterator position, size_t n, parameter_t value)
637  {
638  ETL_ASSERT((size() + n) <= CAPACITY, ETL_ERROR(vector_full));
639 
640  size_t insert_n = n;
641  size_t insert_begin = std::distance(begin(), position);
642  size_t insert_end = insert_begin + insert_n;
643 
644  // Copy old data.
645  size_t copy_old_n;
646  size_t construct_old_n;
647  iterator p_construct_old;
648 
649  if (insert_end > size())
650  {
651  copy_old_n = 0;
652  construct_old_n = size() - insert_begin;
653  p_construct_old = p_buffer + insert_end;
654  }
655  else
656  {
657  copy_old_n = size() - insert_begin - insert_n;
658  construct_old_n = insert_n;
659  p_construct_old = p_end;
660  }
661 
662  size_t copy_new_n = construct_old_n;
663  size_t construct_new_n = insert_n - copy_new_n;
664 
665  // Construct old.
666  etl::uninitialized_copy_n(p_end - construct_old_n, construct_old_n, p_construct_old);
667  ETL_ADD_DEBUG_COUNT(construct_old_n)
668 
669  // Copy old.
670  etl::copy_n(p_buffer + insert_begin, copy_old_n, p_buffer + insert_end);
671 
672  // Construct new.
673  etl::uninitialized_fill_n(p_end, construct_new_n, value);
674  ETL_ADD_DEBUG_COUNT(construct_new_n)
675 
676  // Copy new.
677  std::fill_n(p_buffer + insert_begin, copy_new_n, value);
678 
679  p_end += n;
680  }
