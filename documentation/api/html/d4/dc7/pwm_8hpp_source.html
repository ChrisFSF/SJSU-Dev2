73  void Initialize(uint32_t frequency_hz = 1'000) override
74  {
75  SJ2_ASSERT_FATAL(1 <= channel_ && channel_ <= 6,
76  "Channel must be between 1 and 6.");
77  // Enables PWM1 power/clock control bit
78  // TODO(#): Replace direct manipulation of system clock register.
79  sc->PCONP |= SysClk::kActivatePwm1;
80  // Resets PWMTC on Match with MR0
81  pwm1->MCR |= PwmConfigure::kResetMr0;
82  pwm1->MR0 = config::kSystemClockRate / frequency_hz;
83  // Enables PWM TC and PC for counting and enables PWM mode
84  EnablePwm();
85  pwm1->CTCR &= ~PwmConfigure::kTimerMode;
86  // Enables PWM[channel] output
87  pwm1->PCR |= PwmOutputEnable(channel_);
88  pwm_->SetPinFunction(kPwmFunction);
89  }
