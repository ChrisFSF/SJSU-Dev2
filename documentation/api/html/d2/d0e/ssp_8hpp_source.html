183  void Initialize() override
184  {
185  uint32_t pssp = static_cast<uint32_t>(pssp_);
186 
187  DataSize check_data_size;
188  if (frame_format_ == kMicro)
189  {
190  check_data_size = kEight;
191  }
192  else
193  {
194  check_data_size = data_size_;
195  }
196  // Power up peripheral
197  PowerUpPeripheral(kPowerBit[pssp]);
198  ssp_registers[pssp]->CPSR &= ~(0xFF);
199  ssp_registers[pssp]->CPSR |= (clock_prescaler_);
200 
201  // Enable SSP pins
202  mosi_->SetPinFunction(kPinSelect[pssp]);
203  miso_->SetPinFunction(kPinSelect[pssp]);
204  sck_->SetPinFunction(kPinSelect[pssp]);
205 
206  // Clear and Set Control Register values
207  ssp_registers[pssp]->CR0 &=
208  ~((0xF << kDataBit) | (0x3 << kFrameBit) | (0x3 << kPolarityBit) |
209  (0xFF << kDividerBit));
210  ssp_registers[pssp]->CR0 |=
211  (check_data_size << kDataBit) | (frame_format_ << kFrameBit);
212 
213  // Set clk polarity = 1, clk phase set to 0
214  ssp_registers[pssp]->CR0 |=
215  (clock_polarity_ << kPolarityBit) | (clock_phase_ << kPhaseBit);
216  ssp_registers[pssp]->CR0 |= (clock_divider_ << kDividerBit);
217  // Enable SSP
218  ssp_registers[pssp]->CR1 |= (1 << 1);
219  // Set SSP as master or slave
220  ssp_registers[pssp]->CR1 &= ~(1 << kMasterModeBit);
221  ssp_registers[pssp]->CR1 |= (master_mode_ << kMasterModeBit);
222  }
