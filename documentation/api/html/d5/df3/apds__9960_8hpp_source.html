350  void ProcessGestureData(uint8_t level) override
351  {
352  // Read Gesture FIFOs (U/D/L/R)
353  // Store raw gesture data in a 4x32 uint8_t array
354  for (int i = 0; i < (4 * level); i += 4)
355  {
356  // Check which photodiode pair sensed more IR light
357  if (abs((gfifo_data_[i] - gfifo_data_[i + 1])) >
358  abs((gfifo_data_[i + 2] - gfifo_data_[i + 3])))
359  {
360  // If a photodiode has a smaller value,
361  // then that photodiode was covered and movement detected
362  if ((gfifo_data_[i] - gfifo_data_[i + 1]) < up_sensitivity_) // Up
363  {
364  points_[index_] = Direction::kUp;
365  index_++;
366  }
367  else if ((gfifo_data_[i] - gfifo_data_[i + 1]) >
368  down_sensitivity_) // Down
369  {
370  points_[index_] = Direction::kDown;
371  index_++;
372  }
373  }
374  else
375  {
376  // If a photodiode has a smaller value,
377  // then that photodiode was covered and movement detected
378  if ((gfifo_data_[i + 2] - gfifo_data_[i + 3]) <
379  left_sensitivity_) // Left
380  {
381  points_[index_] = Direction::kLeft;
382  index_++;
383  }
384  else if ((gfifo_data_[i + 2] - gfifo_data_[i + 3]) >
385  right_sensitivity_) // Right
386  {
387  points_[index_] = Direction::kRight;
388  index_++;
389  }
390  }
391  // Check the sum of all photodiodes
392  if ((gfifo_data_[i] + gfifo_data_[i + 1] + gfifo_data_[i + 2] +
393  gfifo_data_[i + 3]) > near_sensitivity_) // Near
394  {
395  near_count_++;
396  }
397  else if ((gfifo_data_[i] + gfifo_data_[i + 1] + gfifo_data_[i + 2] +
398  gfifo_data_[i + 3]) < far_sensitivity_) // Far
399  {
400  far_count_++;
401  }
402  }
403  }
404  Gesture DecodeGestureData(uint8_t level) override
405  {
406  Gesture result = kError;
407 
408  if (level > 15)
409  {
410  if (far_count_ > 5)
411  {
412  result = kFAR;
413  }
414  else if (near_count_ > 5)
415  {
416  result = kNEAR;
417  }
418  }
419 
420  for (int i = 0; i < kMaxDataSize; i++)
421  {
422  // Check initial point, then check next point
423  // to figure out the direction of the swipe
424  switch (points_[i])
425  {
426  case Direction::kUp:
427  if (points_[i + 1] == Direction::kDown)
428  {
429  result = kSwipeDOWN;
430  }
431  break;
432  case Direction::kDown:
433  if (points_[i + 1] == Direction::kUp)
434  {
435  result = kSwipeUP;
436  }
437  break;
438  case Direction::kLeft:
439  if (points_[i + 1] == Direction::kRight)
440  {
441  result = kSwipeRIGHT;
442  }
443  break;
444  case Direction::kRight:
445  if (points_[i + 1] == Direction::kLeft)
446  {
447  result = kSwipeLEFT;
448  }
449  break;
450  // If no valid point, then check near/far counts
451  default: break;
452  }
453  }
454  return result;
455  }
456  Gesture GetGesture() override
457  {
458  if (CheckIfGestureOccured())
459  {
460  // check if gesture ended
461  if (ReadGestureMode() == 0)
462  {
463  uint8_t level = 0;
464  level = (GetGestureFIFOLevel());
465  ReadGestureFIFO(gfifo_data_, level);
466 
467  // reset processing variables
