251  uint32_t CalculatePll(PllInput input_frequency, uint16_t desired_speed_in_mhz)
252  {
253  SJ2_ASSERT_FATAL(desired_speed_in_mhz < 384 && desired_speed_in_mhz > 12,
254  "Frequency must be lower than 384 MHz"
255  "and greater than or equal to 12 MHz");
256  bool calculating = true;
257  uint32_t multiplier_value;
258  if ((desired_speed_in_mhz % static_cast<uint16_t>(input_frequency)) >= 1)
259  {
260  multiplier_value = static_cast<uint32_t>(
261  (desired_speed_in_mhz / static_cast<uint16_t>(input_frequency)) + 1);
262  }
263  else
264  {
265  multiplier_value = static_cast<uint32_t>(
266  desired_speed_in_mhz / static_cast<uint16_t>(input_frequency));
267  }
268  uint16_t divider_value = 1;
269  while (calculating)
270  {
271  uint16_t current_controlled_oscillator_frequency;
272  current_controlled_oscillator_frequency = static_cast<uint16_t>(
273  (static_cast<uint16_t>(input_frequency) * multiplier_value * 2) *
274  divider_value);
275  if (current_controlled_oscillator_frequency >= 156)
276  {
277  calculating = false;
278  }
279  else
280  {
281  divider_value = static_cast<uint16_t>(divider_value * 2);
282  SJ2_ASSERT_FATAL(divider_value < 8,
283  "PLL divider value went out of bounds");
284  }
285  }
286 
287  return multiplier_value;
288  }

290  uint32_t SetMainPll(PllInput input_frequency, uint16_t desired_speed_in_mhz)
291  {
292  uint16_t divider_value = 1;
293  uint64_t timeout_time = Milliseconds() + kDefaultTimeout;
294  uint64_t current_time = Milliseconds();
295  uint32_t multiplier_value =
296  CalculatePll(input_frequency, desired_speed_in_mhz);
297  uint32_t actual_speed =
298  static_cast<uint32_t>(input_frequency) * multiplier_value;
299  // TO DO: use registers to retreive values
300  SelectOscillatorSource(OscillatorSource::kIrc);
301  SelectMainClockSource(MainClockSource::kBaseClock);
302  SelectUsbClockSource(UsbSource::kBaseClock);
303  SelectSpifiClockSource(SpifiSource::kBaseClock);
304  // must subtract 1 from multiplier value as specified in datasheet
305  system_controller->PLL0CFG =
306  (system_controller->PLL0CFG & ~kClearPllMultiplier) |
307  (multiplier_value - 1);
308  system_controller->PLL0CFG =
309  (system_controller->PLL0CFG & ~kClearPllDivider) | (divider_value << 5);
310  system_controller->PLL0CON |= kEnablePll;
311  // nessecary feed sequence to ensure the changes are intentional
312  system_controller->PLL0FEED = 0xAA;
313  system_controller->PLL0FEED = 0x55;
314  while (!(system_controller->PLL0STAT >> kPlock & 1) &&
315  (current_time < timeout_time))
316  {
317  current_time = Milliseconds();
318  }
319  if (!(system_controller->PLL0STAT >> kPlock & 1) &&
320  (current_time >= timeout_time))
321  {
322  SJ2_ASSERT_FATAL(false,
323  "PLL lock could not be established before timeout");
324  actual_speed = kDefaultIRCFrequency;
325  }
326  return (actual_speed - desired_speed_in_mhz);
327  }
