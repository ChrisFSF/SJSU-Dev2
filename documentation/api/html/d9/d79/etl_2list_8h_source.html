1399  void sort(TCompare compare)

1405  iterator i_tail;
1406  int list_size = 1;
1407  int number_of_merges;
1408  int left_size;

1410 
1411  if (is_trivial_list())
1412  {
1413  return;
1414  }
1415 
1416  while (true)
1417  {
1418  i_left = begin();
1419  i_head = end();
1420  i_tail = end();
1421 
1422  number_of_merges = 0; // Count the number of merges we do in this pass.
1423 
1424  while (i_left != end())
1425  {
1426  ++number_of_merges; // There exists a merge to be done.
1427  i_right = i_left;
1428  left_size = 0;
1429 
1430  // Step 'list_size' places along from left
1431  for (int i = 0; i < list_size; ++i)
1432  {
1433  ++left_size;
1434  ++i_right;
1435 
1436  if (i_right == end())
1437  {
1438  break;
1439  }
1440  }
1441 
1442  // If right hasn't fallen off end, we have two lists to merge.
1443  right_size = list_size;
1444 
1445  // Now we have two lists. Merge them.
1446  while (left_size > 0 || (right_size > 0 && i_right != end()))
1447  {
1448  // Decide whether the next node of merge comes from left or right.
1449  if (left_size == 0)
1450  {
1451  // Left is empty. The node must come from right.
1452  i_node = i_right++;
1453  --right_size;
1454  }
1455  else if (right_size == 0 || i_right == end())
1456  {
1457  // Right is empty. The node must come from left.
1458  i_node = i_left++;
1459  --left_size;
1460  }
1461  else if (!compare(*i_right, *i_left))
1462  {
1463  // First node of left is lower or same. The node must come from left.
1464  i_node = i_left++;
1465  --left_size;
1466  }
1467  else
1468  {
1469  // First node of right is lower. The node must come from right.
1470  i_node = i_right;
1471  ++i_right;
1472  --right_size;
1473  }
1474 
1475  // Add the next node to the merged head.
1476  if (i_head == end())
1477  {
1478  join(*i_head.p_node, *i_node.p_node);
1479  i_head = i_node;
1480  i_tail = i_node;
1481  }
1482  else
1483  {
1484  join(*i_tail.p_node, *i_node.p_node);
1485  i_tail = i_node;
1486  }
1487 
1488  join(*i_tail.p_node, terminal_node);
