657  std::pair<iterator, bool> insert(const value_type& key)
658  {
659  std::pair<iterator, bool> result(end(), false);
660 
661  ETL_ASSERT(!full(), ETL_ERROR(unordered_multiset_full));
662 
663  // Get the hash index.
664  size_t index = get_bucket_index(key);
665 
666  // Get the bucket & bucket iterator.
667  bucket_t* pbucket = pbuckets + index;
668  bucket_t& bucket = *pbucket;
669 
670  // The first one in the bucket?
671  if (bucket.empty())
672  {
673  // Get a new node.
674  node_t& node = *pnodepool->allocate<node_t>();
675  ::new (&node.key) value_type(key);
676  ETL_INCREMENT_DEBUG_COUNT
677 
678  // Just add the pointer to the bucket;
679  bucket.insert_after(bucket.before_begin(), node);
680 
681  result.first = iterator((pbuckets + number_of_buckets), pbucket, pbucket->begin());
682  result.second = true;
683 
684  adjust_first_last_markers(pbucket);
685  }
686  else
687  {
688  // Step though the bucket looking for a place to insert.
689  local_iterator inode_previous = bucket.before_begin();
690  local_iterator inode = bucket.begin();
691 
692  while (inode != bucket.end())
693  {
694  // Do we already have this key?
695  if (inode->key == key)
696  {
697  break;
698  }
699 
700  ++inode_previous;
701  ++inode;
702  }
703 
704  // Get a new node.
705  node_t& node = *pnodepool->allocate<node_t>();
706  ::new (&node.key) value_type(key);
707  ETL_INCREMENT_DEBUG_COUNT
708 
709  // Add the node to the end of the bucket;
710  bucket.insert_after(inode_previous, node);
711  ++inode_previous;
712 
713  result.first = iterator((pbuckets + number_of_buckets), pbucket, inode_previous);
714  result.second = true;
715  }
716 
717  return result;
718  }
