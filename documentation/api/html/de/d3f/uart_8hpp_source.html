208  UartCalibration_t GenerateUartCalibration(float baud_rate)
209  {
210  States state = States::kCalculateIntegerDivideLatch;
211  UartCalibration_t uart_calibration;
212  float divide_estimate = 0;
213  float decimal = 1.5;
214  float div = 1;
215  float mul = 2;
216  while (state != States::kDone)
217  {
218  switch (state)
219  {
220  case States::kCalculateIntegerDivideLatch:
221  {
222  divide_estimate = DividerEstimate(baud_rate);
223  if (divide_estimate < 1.0f)
224  {
225  uart_calibration.divide_latch = 0;
226  state = States::kDone;
227  }
228  else if (IsDecmial(divide_estimate))
229  {
230  uart_calibration.divide_latch =
231  static_cast<uint32_t>(divide_estimate);
232  state = States::kDone;
233  }
234  else
235  {
236  state = States::kCalculateDivideLatchWithDecimal;
237  }
238  break;
239  }
240  case States::kCalculateDivideLatchWithDecimal:
241  {
242  divide_estimate = roundf(DividerEstimate(baud_rate, decimal));
243  decimal = FractionalEstimate(baud_rate, divide_estimate);
244  if (1.1f <= decimal && decimal <= 1.9f)
245  {
246  state = States::kGenerateFractionFromDecimal;
247  }
248  else
249  {
250  state = States::kDecimalFailedGenerateNewDecimal;
251  }
252  break;
253  }
254  case States::kDecimalFailedGenerateNewDecimal:
255  {
256  mul += 1;
257 
258  if (div > 15)
259  {
260  state = States::kDone;
261  break;
262  }
263  else if (mul > 15)
264  {
265  div += 1;
266  mul = div + 1;
267  }
268  decimal = div / mul;
269  state = States::kCalculateDivideLatchWithDecimal;
270  break;
271  }
272  case States::kGenerateFractionFromDecimal:
273  {
274  uart_calibration = FindClosestFractional(decimal);
275  uart_calibration.divide_latch =
276  static_cast<uint32_t>(divide_estimate);
277  state = States::kDone;
278  break;
279  }
280  case States::kDone: { break;
281  }
282  default: { break;
283  }
284  }
285  }
286  return uart_calibration;
287  }
