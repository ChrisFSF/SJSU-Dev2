168  bool push_implementation(parameter_t value)
169  {
170  if (current_size != MAX_SIZE)
171  {
172  ::new (&p_buffer[write_index]) T(value);
173 
174  write_index = get_next_index(write_index, MAX_SIZE);
175 
176  ++current_size;
177 
178  return true;
179  }
180 
181  // Queue is full.
182  return false;
183  }

185  //*************************************************************************
187  //*************************************************************************
188  bool pop_implementation(reference value)
189  {
190  if (current_size == 0)
191  {
192  // Queue is empty
193  return false;
194  }
195 
196  value = p_buffer[read_index];
197  p_buffer[read_index].~T();
198 
199  read_index = get_next_index(read_index, MAX_SIZE);
200 
201  --current_size;
202 
203  return true;
