411  // TODO(#344):
412  // If this Send() function gets a pointer that only has scope within a
413  // function call, then once that function goes out of scope, the pointer will
414  // become a dangling pointer (pointer to invalid data). Afterwards, when
415  // WriteMessageToBuffer() tries to access the data to write it to the HW
416  // buffers, it will get corrupt data. This problem can arise if the user wants
417  // to send a 1 time message. However, if the user is sending cyclic messages,
418  // it is assumed that global message handles exist somewhere in the user's app
419  // layer since they need to be updated periodically.
420  bool Send(TxMessage_t * const kMessage, uint32_t id,
421  const uint8_t * const kPayload, size_t length) const override
422  {
423  bool success = false;
424 
425  // Capture status flag register
426  status[controller_].SR = can_registers[controller_]->SR;
427 
428  // Assemble the CAN message with the user provided data.
429  //
430  // Since this memory location is written to by this function and is read by
431  // the ISR (to eventually write the data at this memory location into the
432  // transmit buffer), we disable the CAN interrupt to ensure that complete
433  // and valid data is written before it is accessed by the ISR.
434  DeregisterIsr(CAN_IRQn);
435  kMessage->id = id;
436  kMessage->frame_data.data_length = (length & 0xF);
437  kMessage->frame_data.remote_tx_request = 0;
438  for (uint8_t i = 0; i < length; i++)
439  {
440  kMessage->data.bytes[i] = kPayload[i];
441  }
442  RegisterIsr(CAN_IRQn, &CanIrqHandler, true, 5);
443 
444  // Check if any buffer is available.
445  if (status[controller_].flags.tx_buffer_1_released)
446  {
447  WriteMessageToBuffer(kMessage, kBuffer1, controller_);
448  success = true;
449  }
450  else if (status[controller_].flags.tx_buffer_2_released)
451  {
452  WriteMessageToBuffer(kMessage, kBuffer2, controller_);
453  success = true;
454  }
455  else if (status[controller_].flags.tx_buffer_3_released)
456  {
457  WriteMessageToBuffer(kMessage, kBuffer3, controller_);
458  success = true;
459  }
460  else
461  {
462  // All buffers busy, enqueue the message pointer.
463  // TODO(#345): Ideally this should be a priority queue based on message
464  // ID.
465  if (xQueueSend(transmit_queue[controller_], &kMessage, 0) == pdPASS)
466  {
467  success = true;
468  }
469  }
470  return success;
471  }

484  bool SelfTestBus(uint32_t id) override
485  {
486  bool success = false;
487 
488  TxMessage_t test_message_tx;
489  RxMessage_t test_message_rx;
490 
491  memset(&test_message_tx, 0, sizeof(TxMessage_t));
492  memset(&test_message_rx, 0, sizeof(RxMessage_t));
493 
494  test_message_tx.id = id;
495 
496  // Enable reset mode in order to write to registers
497  SetControllerMode(kReset, true);
498  // Enable self-test mode
499  SetControllerMode(kSelfTest, true);
500  // Disable reset mode
501  SetControllerMode(kReset, false);
502  // Write test message to tx buffer 1
503  WriteMessageToBuffer(&test_message_tx, kBuffer1, controller_);
504  // Set self-test request and send buffer 1
505  can_registers[controller_]->CMR = kSelfReceptionSendTxBuffer1;
506 
507  // Wait for HW
508  Delay(1);
509 
510  // Get the message; the ISR (interrupt service routine)
511  // will read the message from the rx buffer
512  // and enqeue it into the rx queue.
513  Receive(&test_message_rx);
514 
515  // Check if the received message matches the one we sent
516  if (test_message_rx.id == test_message_tx.id)
517  {
518  success = true;
519  }
520 
521  // Disable self-test mode
522  SetControllerMode(kReset, true);
523  SetControllerMode(kSelfTest, false);
524  SetControllerMode(kReset, false);
525 
526  return success;
527  }

566  [[gnu::always_inline]] static void ProcessIrq()
567  {
568  TxMessage_t * message_ptr;
569  bool transmit_interrupt, receive_interrupt;
570 
571  for (uint8_t controller = 0; controller < kNumberOfControllers;
572  controller++)
573  {
574  if (is_controller_initialized[controller] == true)
575  {
576  // Capture interrupt register
577  // Bits are cleared when you read the ICR register
578  interrupts[controller].ICR = can_registers[controller]->ICR;
579 
580  transmit_interrupt = interrupts[controller].flags.tx_buffer_1_ready ||
581  interrupts[controller].flags.tx_buffer_2_ready ||
582  interrupts[controller].flags.tx_buffer_3_ready;
583 
584  receive_interrupt = interrupts[controller].flags.rx_buffer_full;
585 
586  if (transmit_interrupt == true)
587  {
588  if (xQueueIsQueueEmptyFromISR(transmit_queue[controller]) == pdFALSE)
589  {
590  // dequeue message pointer
591  xQueueReceiveFromISR(transmit_queue[controller], &message_ptr, 0);
592  // send it to the available buffer
593  if (interrupts[controller].flags.tx_buffer_1_ready)
594  {
595  WriteMessageToBuffer(message_ptr, kBuffer1,
596  static_cast<Controllers>(controller));
597  }
598  else if (interrupts[controller].flags.tx_buffer_2_ready)
599  {
600  WriteMessageToBuffer(message_ptr, kBuffer2,
601  static_cast<Controllers>(controller));
602  }
603  else if (interrupts[controller].flags.tx_buffer_3_ready)
604  {
605  WriteMessageToBuffer(message_ptr, kBuffer3,
606  static_cast<Controllers>(controller));
607  }
608  else
609  {
610  // will never occur
611  }
612  }
613  }
614 
615  if (receive_interrupt == true)
616  {
617  ReadMessageFromBuffer(static_cast<Controllers>(controller));
618  }
619  }
620  }
621  }

644  [[gnu::always_inline]] static void WriteMessageToBuffer(
645  const TxMessage_t * const kMessage, TxBuffers buffer_number,
646  Controllers controller)
647  {
648  volatile TxMessage_t * ptr_to_buffer_offset;
649  Commands command;
650 
651  switch (buffer_number)
652  {
653  case kBuffer1:
654  {
655  ptr_to_buffer_offset = reinterpret_cast<volatile TxMessage_t *>(
656  &can_registers[controller]->TFI1);
657  command = kSendTxBuffer1;
658  }
659  break;
660  case kBuffer2:
661  {
662  ptr_to_buffer_offset = reinterpret_cast<volatile TxMessage_t *>(
663  &can_registers[controller]->TFI2);
664  command = kSendTxBuffer2;
665  }
666  break;
667  case kBuffer3:
668  {
669  ptr_to_buffer_offset = reinterpret_cast<volatile TxMessage_t *>(
670  &can_registers[controller]->TFI3);
671  command = kSendTxBuffer3;
672  }
673  break;
674  default:
675  // should never occur
676  break;
677  }
678 
679  DeregisterIsr(CAN_IRQn);
680  ptr_to_buffer_offset->TFI = kMessage->TFI;
681  ptr_to_buffer_offset->id = kMessage->id;
682  ptr_to_buffer_offset->data.qword = kMessage->data.qword;
683  can_registers[controller]->CMR = command;
684  RegisterIsr(CAN_IRQn, &CanIrqHandler, true, 5);
685  }

787  // http://www.keil.com/dd/docs/datashts/silabs/boschcan_ug.pdf
788  //

790  //

794  //

796  // | TSCL | TSEG1 | TSEG2 | LPC
797  // ^
798  // sample point (industry standard: 80%)
