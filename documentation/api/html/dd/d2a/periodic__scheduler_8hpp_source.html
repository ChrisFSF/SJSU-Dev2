110 
111  PeriodicScheduler()
112  : Task("Periodic Scheduler", Priority::kHigh),
113  task_list_{ nullptr },
114  counters_{ 0 }
115  {
116  // The delay time should be 1 since the FreeRTOS tick rate is set to
117  // 1 tick = 1000Hz and 1000Hz is fastest rate for the PeriodicScheduler
118  SetDelayTime(1);
119  }
120  bool Run() override
121  {
122  // Since the FreeRTOS tick rate is set to 1000Hz, the 1000Hz task should be
123  // executed every 1 tick and the 100Hz, 10Hz, and 1Hz tasks would execute at
124  // 1/10th of the subsequent rates
125  if (HandlePeriodicSemaphore(Frequency::k1000Hz, 1))
126  {
127  if (HandlePeriodicSemaphore(Frequency::k100Hz, 10))
128  {
129  if (HandlePeriodicSemaphore(Frequency::k10Hz, 10))
130  {
131  if (HandlePeriodicSemaphore(Frequency::k1Hz, 10))
132  {
133  // NOLINT
134  }
135  }
136  }
137  }
138  return true;
139  }

175  bool HandlePeriodicSemaphore(Frequency frequency, uint8_t period)
176  {
177  bool semaphore_given = false;
178  if (++counters_[util::Value(frequency)] == period)
179  {
180  counters_[util::Value(frequency)] = 0;
181  PeriodicTaskInterface * task = task_list_[util::Value(frequency)];
182  // do nothing if no task is scheduled for the specified frequency
183  if (task == nullptr)
184  {
185  return true;
186  }
187  SemaphoreHandle_t semaphore = task->GetPeriodicSemaphore();
188  [[maybe_unused]] const char * task_names[] = { "1Hz", "10Hz", "100Hz",
189  "1000Hz" };
190  SJ2_ASSERT_FATAL(xSemaphoreTake(semaphore, 0) == false,
191  "Overrun occured for task with frequency: %s",
192  task_names[util::Value(frequency)]);
193  semaphore_given = true;
194  xSemaphoreGive(semaphore);
195  }
196  return semaphore_given;
197  }
